### setup: 

1. git config --global user.name "user name"

2. git config --global user.email email_address

3. git clone repository_url(https://github.com/msisharif/testGit.git)




### upload command:

1. cd

2. cd /directory

3. cd git(Folder)/

4. git add fileName

5. git status 

6. git commit -m "first commit(any commit sentence)" fileName

7.git push -u origin master


### remote add existing project

1. git init

2. git add .

3. git commit -m "first commit(Ex.)" 

4. git remote add origin(nickname) repository address

5. git push -u origin(nickname) master


###! [rejected] master -> master (fetch first)

1. git fetch origin master:tmp

2. git rebase tmp

3. git push origin HEAD:master

4. git branch -D tmp
